class Solution {
    public int[] solution(int n) {
        int[] answer = new int[(n*(n + 1)) / 2];

        int[][] snail = new int[n][n];
        int cnt = 0, x = -1, y = 0;
        
        for(int i = 0; i < n; i++){
            for(int j = i; j < n; j++){
                if(i % 3 == 0) { // ë°‘ìœ¼ë¡œ
                    x++;
                }else if(i % 3 == 1) { // ì˜¤ë¥¸ìª½
                    y++;
                } else if(i % 3 == 2){ // ëŒ€ê°ì„ 
                    x--;
                    y--;
                }
                snail[x][y] = ++cnt;
            }
        }
        
        int idx = 0;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(snail[i][j] == 0) break;
                answer[idx++] = snail[i][j];
            }
        }
        
        
        return answer;
    }
}
