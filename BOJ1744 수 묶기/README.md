        int N = Integer.parseInt(br.readLine());

        int sum = 0;
        int minusCnt = 0;

        for(int i =0 ; i < N; i++){
            num[i] = Integer.parseInt(br.readLine());

        }

        // ë§ˆì´ë„ˆìŠ¤ ë¨¼ì € ì²˜ë¦¬

            idx += 2;
            minusCnt -= 2;

        // ë‚˜ë¨¸ì§€ëŠ” ë‹¤ ë¬¶ì–´ì£¼ë©´ ë¨

    }
}
